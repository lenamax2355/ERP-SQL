
---Find out App/Req_Group from the Report/Request name

select b.APPLICATION_NAME Req_App_Name ,d.REQUEST_GROUP_NAME,a.CONCURRENT_PROGRAM_NAME Short_Name,e.USER_CONCURRENT_PROGRAM_NAME Program,
f.APPLICATION_NAME Unit_App_Name,c.REQUEST_UNIT_TYPE,c.REQUEST_GROUP_ID,c.REQUEST_UNIT_ID     
from apps.FND_CONCURRENT_PROGRAMS a,apps.FND_APPLICATION_TL b,--apps.FND_APPLICATION b,
apps.FND_REQUEST_GROUP_UNITS c,
apps.FND_REQUEST_GROUPS d,apps.FND_CONCURRENT_PROGRAMS_TL e,apps.FND_APPLICATION_TL f
where a.APPLICATION_ID=c.unit_APPLICATION_ID
and a.CONCURRENT_PROGRAM_ID=c.REQUEST_UNIT_ID
--and b.APPLICATION_ID=c.unit_APPLICATION_ID
and c.APPLICATION_ID=d.APPLICATION_ID
and c.REQUEST_GROUP_ID=d.REQUEST_GROUP_ID
and a.APPLICATION_ID=e.APPLICATION_ID
and a.CONCURRENT_PROGRAM_ID=e.CONCURRENT_PROGRAM_ID
and e.USER_CONCURRENT_PROGRAM_NAME='BMC Material Variance by Reason Code Report'
and e.LANGUAGE='US'
and f.LANGUAGE='US'
and f.APPLICATION_ID=c.unit_APPLICATION_ID
and b.LANGUAGE='US'
and b.APPLICATION_ID=c.APPLICATION_ID
order by d.REQUEST_GROUP_NAME


---------Find All requests in a Application/req_Group
select b.APPLICATION_NAME Req_App_Name ,d.REQUEST_GROUP_NAME,a.CONCURRENT_PROGRAM_NAME Short_Name,e.USER_CONCURRENT_PROGRAM_NAME Program,
f.APPLICATION_NAME Unit_App_Name,c.REQUEST_UNIT_TYPE,c.REQUEST_GROUP_ID,c.REQUEST_UNIT_ID     
from apps.FND_CONCURRENT_PROGRAMS a,apps.FND_APPLICATION_TL b,--apps.FND_APPLICATION b,
apps.FND_REQUEST_GROUP_UNITS c,
apps.FND_REQUEST_GROUPS d,apps.FND_CONCURRENT_PROGRAMS_TL e,apps.FND_APPLICATION_TL f
where a.APPLICATION_ID=c.unit_APPLICATION_ID
and a.CONCURRENT_PROGRAM_ID=c.REQUEST_UNIT_ID
--and b.APPLICATION_ID=c.unit_APPLICATION_ID
and c.APPLICATION_ID=d.APPLICATION_ID
and c.REQUEST_GROUP_ID=d.REQUEST_GROUP_ID
and a.APPLICATION_ID=e.APPLICATION_ID
and a.CONCURRENT_PROGRAM_ID=e.CONCURRENT_PROGRAM_ID
--and e.USER_CONCURRENT_PROGRAM_NAME like 'BMC PO%Item Batch Change Request'--like 'BMC%'
and e.LANGUAGE='US'
and f.LANGUAGE='US'
and f.APPLICATION_ID=c.unit_APPLICATION_ID
and b.LANGUAGE='US'
and b.APPLICATION_ID=c.APPLICATION_ID
and b.APPLICATION_NAME like '%Oracle Purchasing%'
and d.REQUEST_GROUP_NAME='All Reports'
order by d.REQUEST_GROUP_NAME
