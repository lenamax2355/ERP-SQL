

------------AP Bank Info
----Bank & Branch
select BANK_NAME,BANK_BRANCH_NAME,DESCRIPTION,BANK_NUMBER,BANK_NUM Branch_Num,BANK_NAME_ALT,BANK_BRANCH_NAME_ALT,BANK_BRANCH_ID 
from apps.AP_BANK_BRANCHES


----Bank /Branch/Account
select e.BANK_NAME,e.BANK_NUMBER,e.BANK_NUM Branch_Num,to_char(a.BANK_ACCOUNT_ID) Bank_Account_ID,a.BANK_ACCOUNT_NAME,a.BANK_ACCOUNT_NUM,CURRENCY_CODE,a.DESCRIPTION,
b.SEGMENT3 Asset_Acc,c.SEGMENT3 Gain_Acc,d.SEGMENT3 Loss_Acc,
a.INACTIVE_DATE,a.MULTI_CURRENCY_FLAG,a.ZERO_AMOUNTS_ALLOWED,a.RECEIPT_MULTI_CURRENCY_FLAG
from apps.AP_BANK_ACCOUNTS_all a,apps.GL_CODE_COMBINATIONs b,apps.GL_CODE_COMBINATIONs c,apps.GL_CODE_COMBINATIONs d
,apps.AP_BANK_BRANCHES e
where a.ASSET_CODE_COMBINATION_ID=b.CODE_COMBINATION_ID
and a.GAIN_CODE_COMBINATION_ID=c.CODE_COMBINATION_ID 
and a.loss_CODE_COMBINATION_ID=d.CODE_COMBINATION_ID
and a.BANK_BRANCH_ID=e.BANK_BRANCH_ID
order by e.BANK_NAME
