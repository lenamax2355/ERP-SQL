
-----------Monthly Payment of a bank account (from it's check)

select a.ACCOUNT_TYPE,e.BANK_NAME,m.BANK_ACCOUNT_ID,m.BANK_ACCOUNT_ID,m.BANK_ACCOUNT_NAME,a.BANK_ACCOUNT_NAME,m.BANK_ACCOUNT_NUM,a.BANK_ACCOUNT_NUM,CHECK_NUMBER,
m.CHECK_DATE,AMOUNT,m.CURRENCY_CODE,EXCHANGE_RATE,m.BASE_AMOUNT, 
PAYMENT_TYPE_FLAG,m.DOC_CATEGORY_CODE,STATUS_LOOKUP_CODE
CHECK_STOCK_ID
from apps.AP_BANK_ACCOUNTS_all a,apps.AP_BANK_BRANCHES e,apps.AP_CHECKS_all m--,apps.AP_CHECK_STOCKS_all f
where m.BANK_ACCOUNT_ID=a.BANK_ACCOUNT_ID
and a.BANK_BRANCH_ID=e.BANK_BRANCH_ID
--and f.BANK_ACCOUNT_ID=a.BANK_ACCOUNT_ID
and m.CHECK_DATE >= to_date('2008-7-01','yyyy-mm-dd')
and m.CHECK_DATE < to_date('2008-8-01','yyyy-mm-dd')
and e.BANK_NAME like '%Dummy%'
and a.ATTRIBUTE15 is null
order by m.CHECK_DATE
