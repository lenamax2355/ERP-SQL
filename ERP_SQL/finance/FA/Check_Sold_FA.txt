

select a.ASSET_NUMBER,a.ASSET_NUMBER_DESC 
,a.SEGMENT1||'.'||a.SEGMENT2||'.'||a.SEGMENT3 as Category
,a.COMMENTS,b.ATTRIBUTE1 as FlexField
from apps.FA_TRANSACTION_HISTORY_TRX_V a ,apps.FA_ADDITIONS_V b
where TRANSACTION_TYPE='Full Retirement' 
and b.ASSET_ID=a.ASSET_ID
and PERIOD_ENTERED='FEB-04' 
and BOOK_TYPE_CODE='BMC_FA_BOOK' 
and a.ASSET_ID=100003257





-----FA User's work load:

select count(a.TRANSACTION_TYPE_CODE) as FA_Trans_Lines,USER_NAME,a.TRANSACTION_TYPE_CODE as Trans_Type
from apps.FA_TRANSACTION_HISTORY_TRX_V a,apps.FND_USER b
where a.LAST_UPDATED_BY=b.user_id
and PERIOD_ENTERED='MAY-04'
group by USER_NAME,a.TRANSACTION_TYPE_CODE





List FA Trans:

select TRANSACTION_TYPE_CODE,BOOK_TYPE_CODE,TRANSACTION_TYPE,ASSET_NUMBER,ASSET_NUMBER_DESC,TRANSACTION_DATE_ENTERED,PERIOD_COUNTER,PERIOD_ENTERED,COMMENTS
 from apps.FA_TRANSACTION_HISTORY_TRX_V
