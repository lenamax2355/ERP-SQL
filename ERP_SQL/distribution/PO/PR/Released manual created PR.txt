
------MRP released PR (action_history ACTION_CODE have 'IMPORT')
select REQUISITION_HEADER_ID,SEGMENT1 PR_num from apps.PO_REQUISITION_HEADERS_ALL 
where ORG_ID=127 and REQUISITION_HEADER_ID in --exists
(select OBJECT_ID from apps.PO_ACTION_HISTORY HI where OBJECT_TYPE_CODE='REQUISITION'
and ACTION_CODE='IMPORT')

(select OBJECT_ID from apps.PO_ACTION_HISTORY HI where OBJECT_TYPE_CODE='REQUISITION'

select pr.SEGMENT1 PR_num,hi.ACTION_CODE from apps.PO_REQUISITION_HEADERS_ALL pr, apps.PO_ACTION_HISTORY HI
where PR.REQUISITION_HEADER_ID=HI.OBJECT_ID and hi.OBJECT_TYPE_CODE='REQUISITION'
--and hi.ACTION_CODE='IMPORT' 
and pr.SEGMENT1='46'
order by pr.SEGMENT1,hi.SEQUENCE_NUM
