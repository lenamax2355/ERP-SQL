
----------SO line,Customer_Item,Sys_Item, ITEM_IDENTIFIER_TYPE (Internal Item or Customer Item)

select to_char(b.ORDER_NUMBER) OR_Num,
to_Char(a.LINE_NUMBER) Line_Num
,a.FLOW_STATUS_CODE,a..ITEM_IDENTIFIER_TYPE,
,a.ORDERED_ITEM,c.segment1 Sys_Item
,a.SHIPPED_QUANTITY
,a.ORDERED_QUANTITY
,a.SHIPPING_QUANTITY,a.UNIT_SELLING_PRICE,a.UNIT_SELLING_PRICE*a.SHIPPING_QUANTITY*8.28 Amount
,a.CREATION_DATE
,a.LAST_UPDATE_DATE,
a.ACTUAL_SHIPMENT_DATE,a.FULFILLMENT_DATE,
a.TAX_DATE,a.PRICING_DATE,a.SCHEDULE_SHIP_DATE,a.SCHEDULE_ARRIVAL_DATE,a.ACTUAL_ARRIVAL_DATE,a.SCHEDULE_STATUS_CODE
from apps.OE_ORDER_lineS_all a,apps.OE_ORDER_HEADERS_all b,apps.mtl_system_items_b c
where a.HEADER_ID=b.HEADER_ID
and c.INVENTORY_ITEM_ID=a.INVENTORY_ITEM_ID
and a.ship_from_org_id=c.organization_id
and b.ORDER_NUMBER='111715'


----------SO, line, Internal Item or Customer Item

select a.ORDER_NUMBER,b.LINE_NUMBER,b.TAX_CODE,b.CREATION_DATE,b.ORDERED_ITEM,b.ITEM_IDENTIFIER_TYPE,b.ORDERED_ITEM_ID,b.INVENTORY_ITEM_ID,b.TAX_EXEMPT_FLAG 
from apps.OE_ORDER_HEADERS_all a,apps.OE_ORDER_lineS_all b
where a.header_id=b.header_id
and a.ORDER_NUMBER in ('2800294','2800292','2800293')
order by a.ORDER_NUMBER,b.LINE_NUMBER
